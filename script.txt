use [[IPC2]]Proyecto]

create table pais
(
	id_pais int primary key identity,
	pais varchar(200)
)

create table usuarios
(
	id int primary key identity,
	nombres varchar(200) not null,
	apellidos varchar(200) not null,
	nombre_usuario varchar(150) not null,
	pass varchar(100) not null,
	fecha_nacimiento date not null,
	id_pais int not null,
	CONSTRAINT FK_pais FOREIGN KEY (id_pais) REFERENCES pais (id_pais),
	correo varchar(200) not null,
)
create table reportes
(
	id int primary key identity,
	id_usuario int not null,
	CONSTRAINT FK_usuario FOREIGN KEY (id_usuario) REFERENCES usuarios (id),
	tipo_partida varchar(100) not null,
	categoria varchar(100) not null,
	movimientos int not null,
)
create table reportes_torneos
(
	id int primary key identity,
	id_usuario int not null,
	CONSTRAINT FK_usuario FOREIGN KEY (id_usuario) REFERENCES usuarios (id),
	partidas_jugadas int not null,
	juegos_ganados int not null,
	cantidad_jugadores int not null,
)
select * from reportes
select * from usuarios
select * from reportes_torneos
select * from paises


DROP TABLE reportes
drop table usuarios
